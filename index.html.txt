    <script>
        function revelarMensaje() {
            const mensaje = document.getElementById('mensaje-secreto');
            const boton = document.querySelector('button');
            
            if (mensaje.style.display === 'none' || mensaje.style.display === '') {
                mensaje.style.display = 'block';
                boton.textContent = '¡Felicidades!';
            } else {
                mensaje.style.display = 'none';
                boton.textContent = 'Tengo una sorpresa para ti...';
            }
        }
